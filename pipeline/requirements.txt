# Training dependencies for LoRA fine-tuning the default base (Qwen3.6-35B-A3B)
# on a CUDA GPU. Install on the pod (PyTorch 2.x / CUDA 12 image; torch is
# preinstalled there).
#
#   pip install -r requirements.txt
#
# Qwen3.6 needs a recent transformers (qwen3_5_moe hybrid MoE arch). Default is
# bf16 LoRA (`--quant none`), which fits one 80GB H100 (~74GB) — 4-bit QLoRA is
# discouraged for this MoE family. Unsloth is the preferred loader (handles the
# multimodal text-only path, MoE kernels, and the DeltaNet LoRA targets).

transformers>=5.7.0
trl
peft
datasets
accelerate

# Preferred loader for Qwen3.6 (bf16 LoRA) and the legacy --quant 4bit path.
# train.py falls back to transformers + PEFT if Unsloth can't load the arch.
unsloth

wandb

# Fused linear cross-entropy (train.py patches Laguna and Qwen3.5/Qwen3.5-MoE
# forwards so the [seq, vocab] logits tensor is never built). Optional: train.py
# falls back to the standard loss if liger-kernel is unavailable.
liger-kernel

# FP8 (compressed-tensors) base loading — required for any --quant fp8 base.
compressed-tensors>=0.14

# Only needed for --quant 4bit (bitsandbytes QLoRA). Discouraged for the Qwen
# MoE family (large quantization error); kept for the legacy paths.
bitsandbytes

# Qwen3.6 GatedDeltaNet (48 linear-attn layers) at 64k requires the FLA fast
# path; without it Unsloth falls back to a torch kernel that OOMs. Pin
# apache-tvm-ffi: 0.1.12 crashes on import with tilelang (see tilelang#2367).
flash-linear-attention
causal-conv1d
apache-tvm-ffi==0.1.11
